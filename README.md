      author = {Clesle, Anna and Ernestus, Maximilian},
      title = {Clestus},
      year = {2019},
      publisher = {GitHub},
      journal = {GitHub repository},
      howpublished = {\url{https://github.com/erniejunior/clestus}},
    }
